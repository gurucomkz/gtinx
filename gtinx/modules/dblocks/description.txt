
GET_STATEMENT:
	"Get" ( RETURN_FIELDS, CONDITIONS, ...)
	
RETURN_FIELDS:
	* | 
	NON_ASTERISK_RETURN_FIELDS
	
NON_ASTERISK_FIELDS:
	NON_ASTERISK_FIELD 
	[, NON_ASTERISK_RETURN_FIELDS]
	
NON_ASTERISK_FIELD:
	"ID" |
	"NAME" |
	"ACTIVE" |
	... |
	PROPERTY_NAME
	
PROPERTY_NAME:
	"PROPERTY_"+PROPERTY_IDENTIFIER
	
PROPERTY_IDENTIFIER:
	PROPERTY_ID | 
	PROPERTY_KEY
	
CONDITIONS:
	"Array(" CONDITION_PAIRS ")"
	
CONDITION_PAIR:
	CONDITION_LVALUE "=>" CONDITION_RVALUE
	
CONDITION_PAIRS:
	CONDITION_PAIR 
	[, CONDITION_PAIRS]
	
CONDITION_LVALUE:
	NON_ASTERISK_FIELD |
	PROPERTY_LINK_VALUE
	
PROPERTY_LINK_VALUE:
	PROPERTY_NAME+"_"+NON_ASTERISK_FIELD
	
CONDITION_RVALUE:
	PRIMITIME |
	PRIMITIVE_ARRAY |
	LOGIC |
	LOGIC_MIX_ARRAY
	
PRIMITIVES:
	PRIMITIVE [, PRIMITIVES]
	
PRIMITIVE_ARRAY:
	"Array(" PRIMITIVES ")"
	
PRIMITIME:
	NUMBER |
	STRING

LOGIC:
	LOGIC_AND |
	LOGIC_NOT
	
LOGIC_AND:
	LOGIC_AND_FUNC ( PRIMITIVES )
	
LOGIC_NOT:
	LOGIC_NOT_FUNC ( PRIMITIVES )
	
LOGIC_MIX_ARRAY:
	"Array(" LOGIC_MIX_LIST ")"

LOGIC_MIX:
	PRIMITIME |
	PRIMITIVE_ARRAY |
	LOGIC
	
LOGIC_MIX_LIST:
	LOGIC_MIX 
	[, LOGIC_MIX_LIST]
